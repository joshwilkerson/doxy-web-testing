## Button

- Button labels should use sentence case.
- Plain text links should always be rendered in turquoise.

export function SampleButtonLayout(props) {
  return (
    <div
      style={{
        alignItems: "center",
        display: "flex",
        flexWrap: "wrap",
        gap: "1rem",
      }}
      {...props}
    />
  )
}

export function Avatar() {
  return (
    <img
      src={"https://www.placecage.com/g/100/100"}
      alt={"Nic Cage's Avatar"}
      style={{
        borderRadius: "50%",
        width: "100%",
        height: "100%",
        objectFit: "cover",
      }}
    />
  )
}
