  return (
    <div
      style={{
        backgroundColor: `var(${bgColor})`,
        borderRadius: 8,
        color: `var(${color})`,

        fontWeight: 500,

      <div>

        <div style={{ paddingTop: 4 }}>HSL({colorHsl})</div>
      </div>
      <div style={{ paddingTop: "1rem" }}>{bgColor}</div>

  )
}
